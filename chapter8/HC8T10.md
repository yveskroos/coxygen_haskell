-- HC8T10 : Définition du type Book avec deriving Show

-- Définition du type Book
data Book = Book
    { title  :: String
    , author :: String
    , year   :: Int
    } deriving (Show, Eq)

-- Création d'une instance de Book
monLivre :: Book
monLivre = Book
    { title = "Le Petit Prince"
    , author = "Antoine de Saint-Exupéry"
    , year = 1943
    }

-- Programme principal
main :: IO ()
main = do
    putStrLn "Affichage de l'instance Book :"
    print monLivre
