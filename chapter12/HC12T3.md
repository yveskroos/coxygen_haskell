-- Définition de la fonction factorielle
factorielle :: Int -> Int
factorielle 0 = 1                     -- Cas de base
factorielle n = n * factorielle (n-1) -- Cas récursif

-- Fonction principale de test
main :: IO ()
main = do
    let n = 5
    putStrLn ("La factorielle de " ++ show n ++ " est : " ++ show (factorielle n))
