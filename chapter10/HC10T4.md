-- Définition du type paramétrique
data Box a = Box a
    deriving (Show)

-- Instance de Eq
instance Eq a => Eq (Box a) where
    (Box x) == (Box y) = x == y

-- Exemple d'utilisation
main :: IO ()
main = do
    print $ Box 5 == Box 5        
    print $ Box 5 == Box 10       
    print $ Box "abc" == Box "abc" 
    print $ Box "hi" /= Box "bye"
