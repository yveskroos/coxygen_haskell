-- Fonction applyEffects
applyEffects :: (IO Int, IO Int) -> IO Int
applyEffects (io1, io2) =
    (+) <$> io1 <*> io2 >>= \s -> do
        putStrLn ("Somme = " ++ show s)
        return s

-- Programme principal
main :: IO ()
main = do
    let x = return 5 :: IO Int
    let y = return 7 :: IO Int
    res <- applyEffects (x, y)
    putStrLn ("RÃ©sultat final : " ++ show res)
