-- main.hs

import Control.Applicative (liftA2)

-- Fonction safeProduct
safeProduct :: [Maybe Int] -> Maybe Int
safeProduct = foldr (liftA2 (*)) (Just 1)

-- Programme principal
main :: IO ()
main = do
    print $ safeProduct [Just 2, Just 3, Just 4]  
    print $ safeProduct [Just 5, Nothing, Just 7]  
    print $ safeProduct []
