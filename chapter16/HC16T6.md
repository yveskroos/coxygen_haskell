-- HC16T6 : nième nombre de Fibonacci
fibonacci :: Integer -> Integer
fibonacci 0 = 0
fibonacci 1 = 1
fibonacci n = fibonacci (n - 1) + fibonacci (n - 2)

-- Exemple d'utilisation
main :: IO ()
main = do
    let n = 10
    putStrLn $ "Le " ++ show n ++ "ème nombre de Fibonacci est : " ++ show (fibonacci n)
