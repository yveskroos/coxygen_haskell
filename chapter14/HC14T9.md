{-# LANGUAGE PartialTypeSignatures #-}

module Main where

-- Fonction avec type partiel (joker _)
addNumbers :: _ -> _ -> _
addNumbers x y = x + y

concatStrings :: _ -> _ -> _
concatStrings s1 s2 = s1 ++ s2

main :: IO ()
main = do
    -- Test de addNumbers
    print (addNumbers 3 7)      -- Résultat : 10
    
    -- Test de concatStrings
    putStrLn (concatStrings "Hello, " "world!")  -- Résultat : "Hello, world!"
